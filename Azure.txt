apiVersion: argoproj.io/v1alpha1
kind: ApplicationSet
metadata:
  name: apps
  namespace: argocd
spec:
  generators:
    - git:
        repoURL: https://github.com/your-org/app-gitops.git
        revision: main
        directories:
          - path: microservices/*/*
          - path: frontends/*/*
  template:
    metadata:
      # Example: svc-a-dev OR fe-a-dev
      name: '{{path.basename}}-{{path.dirname}}'
      labels:
        app-type: '{{path[0]}}'  # "microservices" or "frontends"
    spec:
      project: default
      source:
        repoURL: https://github.com/your-org/app-gitops.git
        targetRevision: main
        path: charts/app-chart
        helm:
          valueFiles:
            - ../../{{path}}
      destination:
        server: https://kubernetes.default.svc
        namespace: '{{path.basename}}-{{path.dirname}}'
      syncPolicy:
        automated:
          prune: true
          selfHeal: true
        syncOptions:
          - CreateNamespace=true
