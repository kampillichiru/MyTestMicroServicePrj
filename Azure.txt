apiVersion: argoproj.io/v1alpha1
kind: ApplicationSet
metadata:
  name: elasticsearch-appset
  namespace: argocd
spec:
  generators:
    - list:
        elements:
          - name: dev
            namespace: my-elasticsearch
            valuesPath: elasticsearch/dev/values.yaml
            chartVersion: 1.0.0
  template:
    metadata:
      name: elasticsearch-{{name}}
    spec:
      project: default
      sources:
        # Source 1: Helm chart from Azure Registry
        - repoURL: oci://registry.mycompany.azure/come/gm/charts
          chart: elasticsearch
          targetRevision: '{{chartVersion}}'

        # Source 2: Git repo containing values.yaml
        - repoURL: https://github.com/elasticsearch/dev.git
          targetRevision: main
          ref: values

      destination:
        server: https://kubernetes.default.svc
        namespace: '{{namespace}}'

      syncPolicy:
        automated:
          prune: true
          selfHeal: true

      source:
        helm:
          valueFiles:
            - $values/{{valuesPath}}
